pipeline {
    agent any

    tools {
        maven 'Maven' // Ensure Maven is installed on Jenkins
    }

    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/Uwaisnagarro/Assignment5-6.git', branch: 'main'
            }
        }
        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }
    }
    post {
        success {
            echo 'Build Successful'
        }
        failure {
            echo 'Build Failed'
        }
        always {
            cleanWs()
        }
    }
}
